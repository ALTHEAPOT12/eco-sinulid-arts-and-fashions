      function clearCanvas() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      ctx.save();
      ctx.fillStyle = "#191933";
      ctx.fillRect(0, 0, canvas.width, canvas.height);

      function drawSpiralNebula(colors, count, frame = 0, animate = false) {
      const cx = canvas.width/2, cy = canvas.height/2, maxR = 170;
      for(let i=0; i<count; i++) {
      const t = i / count * 8 * Math.PI + (animate ? frame*0.03 : 0);
      const r = (i / count) * maxR;
      ctx.save();
      ctx.strokeStyle = colors[i % colors.length];
      ctx.globalAlpha = 0.7;
      ctx.lineWidth = 4 - 2.5 * (i / count);
      ctx.beginPath();
      ctx.arc(cx + Math.cos(t)*r, cy + Math.sin(t)*r, 20 + 14*Math.sin(i+t/4), 0, 2*Math.PI);
      ctx.stroke();
      ctx.restore();
      }
      }

      function drawStarburst(colors, count, frame = 0, animate = false) {
      const cx = canvas.width/2, cy = canvas.height/2, len = 155;
      for(let i=0; i<count; i++) {
      const angle = (Math.PI * 2 / count) * i + (animate ? Math.sin(frame*0.03 + i)*0.07 : 0);
      ctx.save();
      ctx.strokeStyle = colors[i % colors.length];
      ctx.globalAlpha = 0.85;
      ctx.lineWidth = 3 - 2 * (i / count);
      ctx.beginPath();
      ctx.moveTo(cx, cy);
      ctx.lineTo(cx + Math.cos(angle) * (len + 18*Math.sin(i+frame/9)), cy + Math.sin(angle) * (len + 18*Math.cos(i-frame/7)));
      ctx.stroke();
      ctx.restore();
      }
      // center core
      ctx.save();
      ctx.globalAlpha = 0.45;
      ctx.beginPath();
      ctx.arc(cx, cy, 45, 0, 2*Math.PI);
      ctx.fillStyle = colors[2];
      ctx.fill();
      ctx.restore();
      }

      function drawOrbits(colors, count, frame = 0, animate = false) {
      const cx = canvas.width/2, cy = canvas.height/2;
      for(let i=1; i<=colors.length; i++) {
      ctx.save();
      ctx.strokeStyle = colors[i-1];
      ctx.globalAlpha = 0.5 + 0.2*(i/colors.length);
      ctx.lineWidth = 3.5 - i;
      ctx.beginPath();
      ctx.arc(cx, cy, 50 + i*32, 0, 2*Math.PI);
      ctx.stroke();
      ctx.restore();
      }
      // planets/threads
      for(let i=0; i<count; i++) {
      const angle = animate
      ? (2 * Math.PI / count) * i + frame*0.007*(i%3+1)
      : Math.random() * Math.PI * 2;
      const r = 65 + 80*(animate ? Math.abs(Math.sin(frame*0.009 + i)) : Math.random());
      ctx.save();
      ctx.fillStyle = colors[i % colors.length];
      ctx.globalAlpha = 0.8;
      ctx.beginPath();
      ctx.arc(cx + Math.cos(angle)*r, cy + Math.sin(angle)*r, 10 + 6*(animate ? (0.6+0.4*Math.sin(frame*0.03+i)) : Math.random()), 0, 2*Math.PI);
      ctx.fill();
      ctx.restore();
      }
      }
